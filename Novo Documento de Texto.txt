function todayKey() {
  const d = new Date();
  return `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`;
}

function easter(y) {
  const a=y%19,b=Math.floor(y/100),c=y%100,d=Math.floor(b/4),e=b%4,
        f=Math.floor((b+8)/25),g=Math.floor((b-f+1)/3),
        h=(19*a+b-d-g+15)%30,i=Math.floor(c/4),k=c%4,
        l=(32+2*e+2*i-h-k)%7,m=Math.floor((a+11*h+22*l)/451),
        mo=Math.floor((h+l-7*m+114)/31)-1,
        dy=((h+l-7*m+114)%31)+1;
  return new Date(y,mo,dy);
}

function liturgicalSeason(date) {
  const y = date.getFullYear();
  const e = easter(y);
  const doy = n => Math.round((n - new Date(n.getFullYear(),0,1)) / 86400000) + 1;
  const d = doy(date), ed = doy(e);
  const ash = ed - 46, pent = ed + 49;
  // Advent: 4th Sunday before Dec 25
  const xmas = new Date(y,11,25);
  const adventStart = new Date(y,11,25 - (xmas.getDay()||7) - 21);
  const advDay = doy(adventStart);

  if (d < advDay && d >= doy(new Date(y,10,27))) {
    // Check if current year's advent hasn't started, look at prev year
  }
  if (d >= advDay || d < doy(new Date(y,0,7))) {
    return { name:'Advento', bg:'#ede7f6', color:'#4a148c' };
  }
  if (d < doy(new Date(y,0,13))) {
    return { name:'Natal', bg:'#fff8e1', color:'#e65100' };
  }
  if (d < ash) return { name:'Tempo Comum', bg:'#e8f5e9', color:'#1b5e20' };
  if (d < ed)  return { name:'Quaresma',    bg:'#f3e5f5', color:'#6a1b9a' };
  if (d <= pent) return { name:'Tempo Pascal', bg:'#fff9c4', color:'#f57f17' };
  return { name:'Tempo Comum', bg:'#e8f5e9', color:'#1b5e20' };
}

function renderPrayer() {
  const prayers = S.get('prayers', {});
  const el = document.getElementById('prayerStreak');
  el.innerHTML = '';
  const now = new Date();
  for (let i = 6; i >= 0; i--) {
    const d = new Date(now); d.setDate(d.getDate() - i);
    const key = `${d.getFullYear()}-${String(d.getMonth()+1).padStart(2,'0')}-${String(d.getDate()).padStart(2,'0')}`;
    const done = !!prayers[key];
    const today = i === 0;
    el.innerHTML += `
      <div class="streak-day">
        <div class="streak-dot ${done?'done':''} ${today?'today':''}">${done?'✓':SDIAS[d.getDay()]}</div>
        <div class="streak-label">${SDIAS[d.getDay()]}</div>
      </div>`;
  }
}

  renderPrayer();
  toast('Oração registrada ✝️');
}

function saveWorkout() {
  if (!selWk) return;
  const w = WK[selWk];
  const exData = w.exercises.map((ex,i)=>({
    name: ex.n,
    sets: Array.from({length:ex.sets},(_,s)=>parseInt(document.getElementById(`ex${i}s${s}`)?.value)||0)
  }));
  const totalReps = exData.reduce((a,ex)=>a+ex.sets.reduce((b,r)=>b+r,0),0);
  const entry = {
    id: Date.now(),
    date: new Date().toISOString(),
    wk: selWk,
    name: w.name,
    exercises: exData,
    totalReps,
    note: document.getElementById('wkNote').value,
  };
  const hist = S.get('workouts', []);
  hist.unshift(entry);
  S.set('workouts', hist);
  document.getElementById('wkForm').style.display = 'none';
  document.querySelectorAll('.wk-btn').forEach(b => b.classList.remove('active'));
  document.getElementById('wkDayHint').textContent = '';
  selWk = null;
  renderWkHistory();
  toast('Treino salvo! 💪');
}

function renderWkHistory() {
  const hist = S.get('workouts', []);
  const el = document.getElementById('wkHistory');
  if (!hist.length) { el.innerHTML = `<div class="empty"><div class="ei">💪</div><p>Nenhum treino registrado ainda.</p></div>`; return; }

    const ds = `${d.getDate()}/${d.getMonth()+1} ${String(d.getHours()).padStart(2,'0')}:${String(d.getMinutes()).padStart(2,'0')}`;

      </div>
      <span class="badge">${e.wk}</span>
    </div>`;
  }).join('');
}

const CAT_COLOR = { lit:'var(--navy)', rel:'var(--forest)', ac:'var(--brown)', out:'var(--muted)' };
const CAT_LABEL = { lit:'Literário', rel:'Espiritual', ac:'Acadêmico', out:'Outro' };

function addBook() {
  const title = document.getElementById('bTitle').value.trim();
  const author = document.getElementById('bAuthor').value.trim();
  if (!title || !author) { toast('Preencha título e autor'); return; }
  const book = {
    id: Date.now(),
    title, author,
    cat: document.getElementById('bCat').value,
    totalPages: parseInt(document.getElementById('bPages').value) || 0,
    currentPage: parseInt(document.getElementById('bCurr').value) || 0,
    status: 'reading',
    startDate: new Date().toISOString(),
    finishedDate: null,
  };
  const books = S.get('books', []);
  books.push(book);
  S.set('books', books);
  closeModal('addBookModal');
  ['bTitle','bAuthor','bPages','bCurr'].forEach(id => document.getElementById(id).value = '');
  renderReading();
  toast('Livro adicionado 📚');
}

function openLogRead(bookId) {
  const books = S.get('books', []);
  const book = books.find(b => b.id === bookId);
  if (!book) return;
  document.getElementById('logReadBody').innerHTML = `
    <div class="field">
      <label style="font-size:14px;font-weight:700;color:var(--text)">${book.title}</label>
      <div style="font-size:12px;color:var(--muted);margin-bottom:12px">${book.author} · Pág. atual: ${book.currentPage}${book.totalPages?' / '+book.totalPages:''}</div>
      <label>Até qual página li hoje? *</label>
      <input type="number" id="lr_page" placeholder="${book.currentPage + 10}" min="${book.currentPage + 1}">
    </div>
    <div class="field">
      <label>Notas / Reflexão (opcional)</label>
      <textarea id="lr_note" placeholder="Trecho marcante, pensamento, dúvida..."></textarea>
    </div>
    <button class="btn btn-navy" onclick="saveSession(${bookId})">Salvar Sessão</button>
  `;
  openModal('logReadModal');
}

function saveSession(bookId) {
  const pageVal = parseInt(document.getElementById('lr_page')?.value);
  const note    = document.getElementById('lr_note')?.value || '';
  const books   = S.get('books', []);
  const book    = books.find(b => b.id === bookId);
  if (!book || isNaN(pageVal) || pageVal <= book.currentPage) { toast('Página inválida'); return; }
  const pagesRead = pageVal - book.currentPage;
  book.currentPage = pageVal;
  if (book.totalPages && pageVal >= book.totalPages) {
    book.status = 'finished';
    book.finishedDate = new Date().toISOString();
  }
  S.set('books', books);
  const sessions = S.get('sessions', []);
  sessions.unshift({ id:Date.now(), bookId, title:book.title, date:new Date().toISOString(), pagesRead, endPage:pageVal, note });
  S.set('sessions', sessions);
  closeModal('logReadModal');
  renderReading();
  toast(`+${pagesRead} páginas! 📖`);
}

function renderReading() {
  const books    = S.get('books', []);
  const sessions = S.get('sessions', []);
  const reading  = books.filter(b => b.status === 'reading');

  // Active books
  const aEl = document.getElementById('activeBooksSection');
  if (!reading.length) {
    aEl.innerHTML = `<div class="section-title">Em Leitura</div><div class="card"><div class="empty"><div class="ei">📖</div><p>Nenhum livro ativo.</p></div></div>`;
  } else {
    aEl.innerHTML = `<div class="section-title">Em Leitura</div>` + reading.map(book => {
      const pct = book.totalPages ? Math.round(book.currentPage / book.totalPages * 100) : null;
      const col = CAT_COLOR[book.cat] || 'var(--navy)';
      return `<div class="card">
        <div class="book-row" style="border:none;padding:0;margin-bottom:12px;">
          <div class="book-spine" style="background:${col}">${book.title.substring(0,14)}</div>
          <div class="book-info">
            <div class="book-title">${book.title}</div>
            <div class="book-author">${book.author}</div>
            <div class="book-status" style="color:${col}">${CAT_LABEL[book.cat]||'Outro'}</div>
            ${pct!==null?`<div class="prog-bar"><div class="prog-fill" style="width:${pct}%;background:${col}"></div></div>
              <div style="font-size:11px;color:var(--muted)">Pág. ${book.currentPage} / ${book.totalPages} (${pct}%)</div>`:
              `<div style="font-size:11px;color:var(--muted)">Pág. atual: ${book.currentPage}</div>`}
          </div>
        </div>
        <button class="btn btn-sm btn-navy" onclick="openLogRead(${book.id})">📝 Registrar Leitura de Hoje</button>
      </div>`;
    }).join('');
  }

  // Sessions history
  const sEl = document.getElementById('sessHistory');
  if (!sessions.length) {
    sEl.innerHTML = `<div class="empty"><div class="ei">📖</div><p>Nenhuma sessão registrada ainda.</p></div>`;
  } else {
    sEl.innerHTML = sessions.slice(0,12).map(s => {
      const d = new Date(s.date);
      return `<div class="hist-item">
        <div>
          <div class="hist-title">${s.title}</div>
          <div class="hist-meta">${d.getDate()}/${d.getMonth()+1}/${d.getFullYear()} · até pág. ${s.endPage}${s.note?' · "'+s.note.substring(0,35)+'…"':''}</div>
        </div>
        <span class="badge">+${s.pagesRead}p</span>
      </div>`;
    }).join('');
  }

  // Library
  const lEl = document.getElementById('library');
  if (!books.length) {
    lEl.innerHTML = `<div class="empty"><div class="ei">🏛️</div><p>Biblioteca vazia.</p></div>`;
  } else {
    lEl.innerHTML = books.map(book => {
      const col = CAT_COLOR[book.cat] || 'var(--navy)';
      return `<div class="book-row">
        <div class="book-spine" style="background:${col}">${book.title.substring(0,14)}</div>
        <div class="book-info">
          <div class="book-title">${book.title}</div>
          <div class="book-author">${book.author}</div>
          <div class="book-status" style="color:${book.status==='finished'?'var(--gold)':col}">
            ${book.status==='finished'?'✓ Concluído':'📖 Lendo'}
          </div>
          ${book.totalPages?`<div style="font-size:11px;color:var(--muted)">
            ${Math.round(book.currentPage/book.totalPages*100)}% · ${book.currentPage}/${book.totalPages} pág.</div>`:''}
        </div>
      </div>`;
    }).join('');
  }
}
